export default function ResultsPage() {
  const [answers, setAnswers] = useState<Record<string, string>>({})
  const [filteredTools, setFilteredTools] = useState<AITool[]>([])
  const [selectedCategory, setSelectedCategory] = useState("all")

  const [viewMode, setViewMode] = useState<"grid" | "list">("grid")
  const [copiedPrompts, setCopiedPrompts] = useState<Record<string, boolean>>({})

  useEffect(() => {
    // Get answers from localStorage
    const storedAnswers = localStorage.getItem("questionnaireAnswers")
    if (storedAnswers) {
      const parsedAnswers = JSON.parse(storedAnswers)
      setAnswers(parsedAnswers)

      // Filter tools based on answers (simplified logic)
      let filtered = [...aiTools]

      // Filter by AI experience level
      if (parsedAnswers.ai_experience === "Complete beginner - explain everything!") {
        filtered = filtered.filter((tool) => tool.difficulty === "Beginner")
      } else if (parsedAnswers.ai_experience === "Some experience with basic tools") {
        filtered = filtered.filter((tool) => tool.difficulty !== "Expert")
      }

      // Handle multiple industries if selected
      if (parsedAnswers.industry && parsedAnswers.industry.includes("Just exploring options")) {
        // Show all tools if they're just exploring
        filtered = [...aiTools]
      }

      setFilteredTools(filtered)
    }
  }, [])

  const categories = ["all", ...Array.from(new Set(filteredTools.map((tool) => tool.category)))]

  const displayedTools =
    selectedCategory === "all" ? filteredTools : filteredTools.filter((tool) => tool.category === selectedCategory)

  const getDifficultyColor = (difficulty: string) => {
    switch (difficulty) {
      case "Beginner":
        return "bg-emerald-100 text-emerald-700 border-emerald-200"
      case "Intermediate":
        return "bg-amber-100 text-amber-700 border-amber-200"
      case "Expert":
        return "bg-rose-100 text-rose-700 border-rose-200"
      default:
        return "bg-gray-100 text-gray-700 border-gray-200"
    }
  }

  const getSpecialBadge = (badge?: string) => {
    switch (badge) {
      case "startup-fav":
        return (
          <Badge className="bg-blue-100 text-blue-700 border-blue-200 flex items-center space-x-1">
            <Zap className="w-3 h-3" />
            <span>Startup Favorite</span>
          </Badge>
        )
      case "creator-pick":
        return (
          <Badge className="bg-purple-100 text-purple-700 border-purple-200 flex items-center space-x-1">
            <UserCheck className="w-3 h-3" />
            <span>Creator's Choice</span>
          </Badge>
        )
      case "our-top-pick":
        return (
          <Badge className="bg-yellow-100 text-yellow-700 border-yellow-200 flex items-center space-x-1">
            <Crown className="w-3 h-3" />
            <span>Our Top Pick</span>
          </Badge>
        )
      case "trending":
        return (
          <Badge className="bg-rose-100 text-rose-700 border-rose-200 flex items-center space-x-1">
            <Award className="w-3 h-3" />
            <span>Trending Now</span>
          </Badge>
        )
      default:
        return null
    }
  }

  const copyPrompt = async (toolId: string, prompt: string) => {
    try {
      // Replace placeholders with user's actual answers
      let personalizedPrompt = prompt
      if (answers.inspiration) {
        personalizedPrompt = personalizedPrompt.replace(/\[BUSINESS TYPE\]/g, answers.industry || "business")
        personalizedPrompt = personalizedPrompt.replace(
          /\[TARGET AUDIENCE\]/g,
          answers.target_audience || "your target audience",
        )
        personalizedPrompt = personalizedPrompt.replace(
          /\[YOUR EXPERTISE\]/g,
          answers.challenge_solving || "your expertise",
        )
      }

      await navigator.clipboard.writeText(personalizedPrompt)
      setCopiedPrompts((prev) => ({ ...prev, [toolId]: true }))
      setTimeout(() => {
        setCopiedPrompts((prev) => ({ ...prev, [toolId]: false }))
      }, 2000)
    } catch (err) {
      console.error("Failed to copy prompt:", err)
    }
  }

  const renderGridCard = (tool: AITool) => (
    <Card
      key={tool.id}
      className="border-rose-100 hover:shadow-xl transition-all duration-300 bg-white/95 backdrop-blur-sm"
    >
      <CardHeader className="pb-4">
        <div className="flex items-start justify-between mb-4">
          <div className="flex items-center space-x-3">
            <CardTitle className="text-2xl font-bold text-gray-800">{tool.name}</CardTitle>
            {tool.isNew && <Badge className="bg-gradient-to-r from-rose-400 to-pink-500 text-white">NEW</Badge>}
          </div>
          <div className="flex items-center space-x-1 bg-yellow-50 px-3 py-1 rounded-full">
            <Star className="w-4 h-4 fill-yellow-400 text-yellow-400" />
            <span className="text-sm font-semibold text-yellow-700">{tool.rating}</span>
          </div>
        </div>

        <div className="flex flex-wrap gap-2 mb-4">
          <Badge className={`${getDifficultyColor(tool.difficulty)} border`}>{tool.difficulty}</Badge>
          <Badge className="bg-gray-50 text-gray-600 border-gray-200 flex items-center space-x-1">
            <DollarSign className="w-3 h-3" />
            <span>{tool.price}</span>
          </Badge>
          {tool.specialBadge && getSpecialBadge(tool.specialBadge)}
        </div>
      </CardHeader>

      <CardContent className="space-y-6">
        <p className="text-gray-600 text-lg leading-relaxed">{tool.description}</p>

        {/* Sign-up requirement */}
        <div className="flex items-center justify-between p-4 bg-gray-50 rounded-xl">
          <span className="font-medium text-gray-700">Account required:</span>
          <Badge className={tool.signUpRequired ? "bg-rose-100 text-rose-700" : "bg-emerald-100 text-emerald-700"}>
            {tool.signUpRequired ? "Yes" : "No"}
          </Badge>
        </div>

        {/* What people use it for */}
        <div className="bg-rose-50 p-6 rounded-xl">
          <h4 className="font-semibold text-gray-800 mb-3 text-lg">Popular uses:</h4>
          <div className="flex flex-wrap gap-2">
            {tool.whatPeopleUseItFor.map((use, index) => (
              <Badge key={index} className="bg-white text-rose-700 border-rose-200 px-3 py-1">
                {use}
              </Badge>
            ))}
          </div>
        </div>

        <div className="bg-blue-50 p-6 rounded-xl">
          <h4 className="font-semibold text-gray-800 mb-3 text-lg flex items-center">
            <Target className="w-5 h-5 mr-2 text-blue-500" />
            Perfect for beginners:
          </h4>
          <p className="text-gray-700 leading-relaxed">{tool.beginnerExplanation}</p>
        </div>

        <div>
          <h4 className="font-semibold text-gray-800 mb-3 text-lg">Key features:</h4>
          <div className="grid grid-cols-2 gap-2">
            {tool.features.map((feature, index) => (
              <div key={index} className="flex items-center space-x-2 text-sm text-gray-600">
                <div className="w-2 h-2 bg-rose-400 rounded-full"></div>
                <span>{feature}</span>
              </div>
            ))}
          </div>
        </div>

        {tool.relatedTools.length > 0 && (
          <div>
            <h4 className="font-semibold text-gray-800 mb-2">Similar tools:</h4>
            <p className="text-gray-600">{tool.relatedTools.join(" • ")}</p>
          </div>
        )}

        {/* Action buttons with copy prompt next to try button */}
        <div className="flex flex-col space-y-3 pt-4">
          <div className="flex space-x-3">
            <Button
              className="flex-1 bg-gradient-to-r from-rose-400 to-pink-500 hover:from-rose-500 hover:to-pink-600 text-white shadow-lg text-lg py-3"
              onClick={() => window.open(tool.website, "_blank")}
            >
              Try {tool.name}
              <ExternalLink className="w-5 h-5 ml-2" />
            </Button>
            <Button
              onClick={() => copyPrompt(tool.id, tool.copyPrompt)}
              className="bg-gradient-to-r from-purple-400 to-pink-400 hover:from-purple-500 hover:to-pink-500 text-white shadow-lg px-6"
            >
              {copiedPrompts[tool.id] ? (
                <>
                  <Check className="w-4 h-4 mr-2" />
                  Copied!
                </>
              ) : (
                <>
                  <Copy className="w-4 h-4 mr-2" />
                  Copy Prompt
                </>
              )}
            </Button>
            <Button className="border-rose-300 text-rose-600 hover:bg-rose-50 bg-transparent p-3" variant="outline">
              <Heart className="w-5 h-5" />
            </Button>
          </div>
          <p className="text-sm text-gray-500 text-center">Ready-to-use prompt personalized for your business</p>
        </div>
      </CardContent>
    </Card>
  )

  const renderListCard = (tool: AITool) => (
    <Card
      key={tool.id}
      className="border-rose-100 hover:shadow-lg transition-all duration-300 bg-white/95 backdrop-blur-sm"
    >
      <CardContent className="p-6">
        <div className="flex items-start justify-between">
          <div className="flex-1">
            <div className="flex items-center space-x-3 mb-3">
              <CardTitle className="text-xl font-bold text-gray-800">{tool.name}</CardTitle>
              {tool.isNew && (
                <Badge className="bg-gradient-to-r from-rose-400 to-pink-500 text-white text-xs">NEW</Badge>
              )}
              <div className="flex items-center space-x-1 bg-yellow-50 px-2 py-1 rounded-full">
                <Star className="w-3 h-3 fill-yellow-400 text-yellow-400" />
                <span className="text-xs font-semibold text-yellow-700">{tool.rating}</span>
              </div>
            </div>

            <div className="flex flex-wrap gap-2 mb-3">
              <Badge className={`${getDifficultyColor(tool.difficulty)} border text-xs`}>{tool.difficulty}</Badge>
              <Badge className="bg-gray-50 text-gray-600 border-gray-200 flex items-center space-x-1 text-xs">
                <DollarSign className="w-2 h-2" />
                <span>{tool.price}</span>
              </Badge>
              {tool.specialBadge && getSpecialBadge(tool.specialBadge)}
            </div>

            <p className="text-gray-600 mb-4 leading-relaxed">{tool.description}</p>

            <div className="flex flex-wrap gap-1 mb-4">
              {tool.whatPeopleUseItFor.slice(0, 3).map((use, index) => (
                <Badge key={index} className="bg-rose-50 text-rose-700 border-rose-200 text-xs px-2 py-1">
                  {use}
                </Badge>
              ))}
              {tool.whatPeopleUseItFor.length > 3 && (
                <Badge className="bg-gray-50 text-gray-600 text-xs px-2 py-1">
                  +{tool.whatPeopleUseItFor.length - 3} more
                </Badge>
              )}
            </div>
          </div>

          <div className="flex flex-col space-y-2 ml-6">
            <Button
              className="bg-gradient-to-r from-rose-400 to-pink-500 hover:from-rose-500 hover:to-pink-600 text-white shadow-lg px-4 py-2"
              onClick={() => window.open(tool.website, "_blank")}
            >
              Try {tool.name}
              <ExternalLink className="w-4 h-4 ml-2" />
            </Button>
            <Button
              onClick={() => copyPrompt(tool.id, tool.copyPrompt)}
              className="bg-gradient-to-r from-purple-400 to-pink-400 hover:from-purple-500 hover:to-pink-500 text-white shadow-lg px-4 py-2"
            >
              {copiedPrompts[tool.id] ? (
                <>
                  <Check className="w-3 h-3 mr-2" />
                  Copied!
                </>
              ) : (
                <>
                  <Copy className="w-3 h-3 mr-2" />
                  Copy Prompt
                </>
              )}
            </Button>
            <Button className="border-rose-300 text-rose-600 hover:bg-rose-50 bg-transparent p-2" variant="outline">
              <Heart className="w-4 h-4" />
            </Button>
          </div>
        </div>
      </CardContent>
    </Card>
  )

  return (
    <div className="container mx-auto py-12">
      <h1 className="text-4xl font-bold text-center text-gray-800 mb-8">AI Tool Recommendations for Your Business</h1>

      {/* Category Tabs */}
      <Tabs defaultValue="all" className="mb-8">
        <TabsList className="justify-center">
          {categories.map((category) => (
            <TabsTrigger
              key={category}
              value={category}
              onClick={() => setSelectedCategory(category)}
              className="data-[state=active]:bg-rose-50 data-[state=active]:text-rose-700"
            >
              {category === "all" ? "All Categories" : category}
            </TabsTrigger>
          ))}
        </TabsList>
      </Tabs>

      {/* View Mode Toggle */}
      <div className="flex items-center justify-end mb-4">
        <span className="mr-3 text-gray-600">View Mode:</span>
        <Button
          variant="outline"
          className={`mr-2 ${viewMode === "grid" ? "bg-rose-50 text-rose-700 border-rose-300" : "border-gray-300 text-gray-600"}`}
          onClick={() => setViewMode("grid")}
        >
          <Grid3X3 className="w-4 h-4 mr-2" />
          Grid
        </Button>
        <Button
          variant="outline"
          className={`${viewMode === "list" ? "bg-rose-50 text-rose-700 border-rose-300" : "border-gray-300 text-gray-600"}`}
          onClick={() => setViewMode("list")}
        >
          <List className="w-4 h-4 mr-2" />
          List
        </Button>
      </div>

      {/* Tools Display */}
      {viewMode === "grid" ? (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8 mb-16">{displayedTools.map(renderGridCard)}</div>
      ) : (
        <div className="space-y-6 mb-16">{displayedTools.map(renderListCard)}</div>
      )}

      {/* Inspirational Section */}
      <div className="bg-gradient-to-r from-rose-400 to-pink-500 rounded-2xl p-12 text-center text-white shadow-xl mb-16">
        <div className="max-w-4xl mx-auto">
          <blockquote className="text-2xl italic mb-6 font-light">
            "The way to get started is to quit talking and begin doing."
          </blockquote>
          <cite className="text-rose-100 text-lg">— Whitney Wolfe Herd, Founder of Bumble</cite>
        </div>
      </div>

      {/* Next Steps */}
      <Card className="border-rose-100 shadow-xl bg-white/95">
        <CardHeader className="text-center pb-8">
          <CardTitle className="text-3xl font-bold text-gray-800 mb-4">Ready to Transform Your Business? 🚀</CardTitle>
          <p className="text-xl text-gray-600">Your success journey starts with the first step</p>
        </CardHeader>
        <CardContent className="space-y-8">
          <div className="grid grid-cols-1 md:grid-cols-3 gap-8">
            <div className="text-center p-6 bg-rose-50 rounded-xl">
              <div className="w-16 h-16 bg-gradient-to-r from-rose-400 to-pink-500 rounded-full flex items-center justify-center mx-auto mb-4 shadow-lg">
                <span className="text-white text-2xl font-bold">1</span>
              </div>
              <h3 className="font-semibold mb-3 text-lg">Start with Your Prompts</h3>
              <p className="text-gray-600">
                Use the personalized prompts we've created specifically for your business vision
              </p>
            </div>

            <div className="text-center p-6 bg-rose-50 rounded-xl">
              <div className="w-16 h-16 bg-gradient-to-r from-rose-400 to-pink-500 rounded-full flex items-center justify-center mx-auto mb-4 shadow-lg">
                <span className="text-white text-2xl font-bold">2</span>
              </div>
              <h3 className="font-semibold mb-3 text-lg">Begin with Beginner Tools</h3>
              <p className="text-gray-600">
                Build confidence with tools marked "Beginner" before advancing to more complex options
              </p>
            </div>

            <div className="text-center p-6 bg-rose-50 rounded-xl">
              <div className="w-16 h-16 bg-gradient-to-r from-rose-400 to-pink-500 rounded-full flex items-center justify-center mx-auto mb-4 shadow-lg">
                <span className="text-white text-2xl font-bold">3</span>
              </div>
              <h3 className="font-semibold mb-3 text-lg">Explore Free Options</h3>
              <p className="text-gray-600">Most tools offer free trials or freemium plans - test them risk-free</p>
            </div>
          </div>

          <div className="flex flex-col sm:flex-row gap-4 justify-center pt-8">
            <Button
              size="lg"
              className="bg-gradient-to-r from-rose-400 to-pink-500 hover:from-rose-500 hover:to-pink-600 text-white px-8 py-4 text-lg shadow-lg"

            >
              Save My Results

            </Button>
            <Link href="/questionnaire">

                variant="outline"
                size="lg"
                className="border-rose-300 text-rose-600 hover:bg-rose-50 bg-transparent px-8 py-4 text-lg"
